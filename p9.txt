import wikipediaapi, ipywidgets as w
from IPython.display import display

wiki = wikipediaapi.Wikipedia(user_agent="MyApp/1.0", language="en")

def fetch(_):
    p = wiki.page(box.value)
    if not p.exists(): print("Not found."); return
    find = lambda k: next((l.split(":")[-1].strip() for l in p.text.split("\n") if k.lower() in l.lower()), "N/A")
    print(f"\n {box.value}\nFounder: {find('Founder')}\nFounded: {find('Founded')}\nBranches: {find('Branches')}\nEmployees: {find('employees')}\nSummary: {p.summary[:400]}")

box = w.Text(placeholder="Enter institution name", description="Institution:")
btn = w.Button(description="Fetch")
btn.on_click(fetch)
display(box, btn)
