import getpass
from langchain_core.prompts import PromptTemplate
from langchain_cohere import ChatCohere
from google.colab import drive

drive.mount('/content/drive')

text = open("/content/Teaching.txt", encoding="utf-8").read()

chain = PromptTemplate(input_variables=["text"], template="""
Summarize and analyze the text below:
{text}
Summary:
Key Takeaways (3 points):
Sentiment (Positive/Negative/Neutral):
""") | ChatCohere(cohere_api_key=getpass.getpass("API Key: "), model="command-r-plus-08-2024")

print(chain.invoke({"text": text}).content)
