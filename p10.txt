from langchain_core.prompts import PromptTemplate
from langchain_cohere import ChatCohere
import wikipediaapi, getpass, ipywidgets as w
from IPython.display import display

llm = ChatCohere(cohere_api_key=getpass.getpass("Cohere API Key: "), model="command-a-03-2025")

wiki = wikipediaapi.Wikipedia(user_agent="IPCBot/1.0", language="en") 
ipc_content = wiki.page("Indian Penal Code").text[:5000]

chain = PromptTemplate(input_variables=["ipc_content", "question"], template="""
You are a legal assistant specialized in the Indian Penal Code.
{ipc_content}
User Question: {question}
Provide a detailed answer and mention the IPC section if applicable.
""") | llm

out = w.Output()

def on_click(_):
    with out:
        out.clear_output()
        if not box.value.strip(): print(" Enter a question."); return
        print(" Fetching...")
        out.clear_output()
        r = chain.invoke({"ipc_content": ipc_content, "question": box.value}).content
        sec = r.split("Section")[1].split(":")[0].strip() if "Section" in r else "N/A"
        print(f" Section: {sec}\n\n Explanation:\n{r.strip()}")

box = w.Text(placeholder="Ask about the Indian Penal Code...", description="You:", layout=w.Layout(width="70%"))
btn = w.Button(description="Ask", button_style="primary", icon="gavel")
btn.on_click(on_click)
display(w.HBox([box, btn]), out)
